local function parseKeyframes(fileContent)
	local Keyframes = {}
	local currentFrame = {}
	for line in fileContent:gmatch("[^\r\n]+") do
		if line:match("%[%d+%]") then
			if currentFrame.Time then table.insert(Keyframes, currentFrame) end
			currentFrame = {}
		elseif line:match("Time: ([%d%.]+)") then
			currentFrame.Time = tonumber(line:match("Time: ([%d%.]+)"))
		elseif line:match("(%w[%w%s]+)%.CFrame = CFrame%.new%(([^%)]+)%)") then
			local partName, cfData = line:match("(%w[%w%s]+)%.CFrame = CFrame%.new%(([^%)]+)%)")
			local args = {}
			for val in cfData:gmatch("[^,]+") do table.insert(args, tonumber(val)) end
			if #args == 12 then currentFrame[partName] = CFrame.new(unpack(args)) end
		end
	end
	if currentFrame.Time then table.insert(Keyframes, currentFrame) end
	return Keyframes
end

local function safeDownload(filepath, url)
	if not isfile(filepath) then
		local success, result = pcall(function()
			return game:HttpGet(url)
		end)
		if success and result then
			writefile(filepath, result)
		end
	end
	downloadedAssets = downloadedAssets + 1
	updateDownloadGui()
end

local getasset = getcustomasset or getsynasset

if getasset and isfile(bgMusicPath) then
	if bgMusicObj then pcall(function() bgMusicObj:Destroy() end) end
	bgMusicObj = Instance.new("Sound")
	bgMusicObj.Name = "BackgroundMusic"
	bgMusicObj.SoundId = getasset(bgMusicPath)
	bgMusicObj.Volume = 0.55
	bgMusicObj.Looped = true
	bgMusicObj.Parent = SoundService
	bgMusicObj:Play()
end

local function fadeOutOrig(snd)
	if not snd then return end
	TS:Create(snd, TweenInfo.new(0.5), {Volume = 0}):Play()
	task.delay(0.5, function() 
		if snd then 
			snd:Stop() 
			snd:Destroy()
		end 
	end)
end

local function fadeInOrig(danceData)
	if not getasset or not isfile(danceData.AudioPath) then return nil end
	if danceData.SoundObj then pcall(function() danceData.SoundObj:Destroy() end) end
	local sound = Instance.new("Sound")
	sound.Name = danceData.Name .. "_NoyaMusic"
	sound.SoundId = getasset(danceData.AudioPath)
	sound.Parent = SoundService
	sound.Volume = 0
	sound.PlaybackSpeed = 1
	sound.Looped = danceData.NoFade or false
	sound.TimePosition = danceData.SkipMusic or 0
	danceData.SoundObj = sound
	sound:Play()
	
	local targetVolume = danceData.Volume or 0.7
	TS:Create(sound, TweenInfo.new(0.5), {Volume = targetVolume}):Play()
	return sound
end

local isDancing = false
local currentDance = nil
local canDance = true
local danceTime, idleTime, walkTime, runTime, jumpTime, fallTime = 0, 0, 0, 0, 0, 0
local isRunning = false
local blendWeight = 1
local loopBlend = 1
local CurrentTransforms = {}

local function giveTools()
	if not isScriptActive then return end
	local backpack = LocalPlayer:FindFirstChildOfClass("Backpack")
	if not backpack then return end
	for _, obj in ipairs(backpack:GetChildren()) do
		if obj:IsA("Tool") and (findDanceData(obj.Name) or obj.Name == "[Char] Run") then obj:Destroy() end
	end
	if currentCharacter then
		for _, obj in ipairs(currentCharacter:GetChildren()) do
			if obj:IsA("Tool") and (findDanceData(obj.Name) or obj.Name == "[Char] Run") then obj:Destroy() end
		end
	end
	
	local runTool = Instance.new("Tool")
	runTool.Name = "[Char] Run"
	runTool.RequiresHandle = false
	table.insert(ToolConnections, runTool.Equipped:Connect(function()
		if not isScriptActive then return end
		isRunning = true
	end))
	table.insert(ToolConnections, runTool.Unequipped:Connect(function()
		if not isScriptActive then return end
		isRunning = false
	end))
	runTool.Parent = backpack

	for _, danceData in ipairs(Dances) do
		local tool = Instance.new("Tool")
		tool.Name = danceData.Name
		tool.RequiresHandle = false
		table.insert(ToolConnections, tool.Equipped:Connect(function()
			if not isScriptActive then return end
			blendWeight = 0
			loopBlend = 1
			if bgMusicObj and bgMusicObj.Parent then
				bgMusicObj:Stop()
			end
			if currentDance then
				local curDanceData = findDanceData(currentDance)
				if curDanceData and curDanceData.SoundObj then
					fadeOutOrig(curDanceData.SoundObj) 
					curDanceData.SoundObj = nil
				end
			end
			currentDance = danceData.Name
			isDancing = true
			danceTime = 0
			fadeInOrig(danceData)
		end))
		table.insert(ToolConnections, tool.Unequipped:Connect(function()
			if not isScriptActive then return end
			if currentDance == danceData.Name then
				isDancing = false
				if danceData.SoundObj then
					fadeOutOrig(danceData.SoundObj)
					danceData.SoundObj = nil
				end
				currentDance = nil
				if bgMusicObj and bgMusicObj.Parent then
					bgMusicObj.TimePosition = 0
					bgMusicObj:Play()
					bgMusicObj.Volume = 0.55
				end
			end
		end))
		tool.Parent = backpack
	end
end

local function setupCharacter(targetChar, isDummy)
	if not isScriptActive or not targetChar then return end
	currentCharacter = targetChar
	fakeHumanoid = targetChar:WaitForChild("Humanoid", 5)
	if not fakeHumanoid then return end
	table.insert(Connections, targetChar.AncestryChanged:Connect(function(_, parent)
		if not parent and hasReanimated and _G.TungTungTungSahur then 
			_G.TungTungTungSahur() 
		end
	end))
	table.insert(Connections, fakeHumanoid.Died:Connect(function()
		if hasReanimated and _G.TungTungTungSahur then 
			_G.TungTungTungSahur() 
		end
	end))
	local oldAnimator = fakeHumanoid:FindFirstChildWhichIsA("Animator")
	if oldAnimator then oldAnimator:Destroy() end
	table.insert(Connections, fakeHumanoid.ChildAdded:Connect(function(child)
		if child:IsA("Animator") then task.defer(function() child:Destroy() end) end
	end))
	rootPart = targetChar:WaitForChild("HumanoidRootPart", 5)
	torso = targetChar:FindFirstChild("Torso") or targetChar:FindFirstChild("UpperTorso")
	if not torso then torso = targetChar:WaitForChild("Torso", 2) end
	headPart = targetChar:FindFirstChild("Head")
	if not headPart then headPart = targetChar:WaitForChild("Head", 2) end
	if rootPart and torso then
		local startTime = tick()
		while tick() - startTime < 3 do
			if rootPart:FindFirstChild("RootJoint") and torso:FindFirstChild("Neck") then break end
			task.wait(0.1)
		end
		CurrentTransforms = {}
		activeJoints = {
			["Torso"] = rootPart:FindFirstChild("RootJoint"),
			["Head"] = torso:FindFirstChild("Neck"),
			["Right Arm"] = torso:FindFirstChild("Right Shoulder"),
			["Left Arm"] = torso:FindFirstChild("Left Shoulder"),
			["Right Leg"] = torso:FindFirstChild("Right Hip"),
			["Left Leg"] = torso:FindFirstChild("Left Hip")
		}
	end
	clearToolConnections()
	isDancing = false
	currentDance = nil
	if bgMusicObj and bgMusicObj.Parent then
		bgMusicObj.TimePosition = 0
		bgMusicObj.Volume = 0.55
		if not bgMusicObj.IsPlaying then bgMusicObj:Play() end
	end
	if isDummy then giveTools() end
end

local function checkAndSetupDummy(char)
	local dummy = char:FindFirstChild("NoyaReanimate") or char:FindFirstChild("FlingReanimate") or char:FindFirstChild("Dummy")
	if not dummy then
		local wsChar = Workspace:FindFirstChild(LocalPlayer.Name)
		if wsChar and wsChar ~= char and wsChar:FindFirstChild("Humanoid") then dummy = wsChar end
	end
	if dummy then
		hasReanimated = true
		setupCharacter(dummy, true)
		return true
	end
	return false
end

local function waitForReanimateAndGiveTools(char)
	task.spawn(function()
		if checkAndSetupDummy(char) then return end
		local found = false
		local childConn
		childConn = Workspace.ChildAdded:Connect(function(child)
			if not isScriptActive then childConn:Disconnect() return end
			if child.Name == LocalPlayer.Name or child.Name == "NoyaReanimate" or child.Name == "FlingReanimate" or child.Name == "Dummy" then
				task.wait() 
				if checkAndSetupDummy(char) then
					found = true
					childConn:Disconnect()
				end
			end
		end)
		for i = 1, 100 do 
			if not isScriptActive or found or not char or not char.Parent then break end
			if checkAndSetupDummy(char) then found = true; childConn:Disconnect(); break end
			task.wait(0.05)
		end
		if not isScriptActive then return end
		if not found then
			hasReanimated = true
			giveTools()
		end
	end)
end

if LocalPlayer.Character then
	originalCharacter = LocalPlayer.Character
	setupCharacter(originalCharacter, false)
	waitForReanimateAndGiveTools(originalCharacter)
end

table.insert(Connections, LocalPlayer.CharacterAdded:Connect(function(char)
	if not isScriptActive then return end
	if hasReanimated then
		if not currentCharacter or not currentCharacter.Parent or (fakeHumanoid and fakeHumanoid.Health <= 0) then
			if _G.TungTungTungSahur then _G.TungTungTungSahur() end
			return
		end
		originalCharacter = char 
		return
	end
	originalCharacter = char
	setupCharacter(char, false)
	waitForReanimateAndGiveTools(char)
end))

table.insert(Connections, RunService.PreSimulation:Connect(function(dt)
	if not isScriptActive or not currentCharacter or not fakeHumanoid or fakeHumanoid.Health <= 0 then return end
	if not canDance or not activeJoints["Torso"] then return end
	fakeHumanoid.WalkSpeed = isRunning and 28 or 16
	local targetCFs = {}
	local lerpSpeed = 1
	if blendWeight < 1 then blendWeight = math.min(1, blendWeight + dt * 4) end
	if loopBlend < 1 then loopBlend = math.min(1, loopBlend + dt * 5) end
	if isDancing and currentDance then
		local danceData = findDanceData(currentDance)
		if danceData and danceData.SoundObj and danceData.SoundObj.Parent and not danceData.NoFade then
			local endT = danceData.EndMusic or (danceData.SoundObj.TimeLength > 0 and danceData.SoundObj.TimeLength - 0.2 or 9e9)
			if danceData.SoundObj.TimePosition >= endT or (danceData.SoundObj.TimeLength > 0 and not danceData.SoundObj.IsPlaying) then
				danceData.SoundObj.TimePosition = danceData.SkipMusic or 0
				if not danceData.SoundObj.IsPlaying then danceData.SoundObj:Play() end
				if danceTime > 0.5 then
					danceTime = 0
					if not danceData.NoLerp then loopBlend = 0 end
				end
			end
		end
		if danceData then
			local Keyframes = danceData.ParsedKeyframes
			if #Keyframes > 0 then
				local prevDanceTime = danceTime
				danceTime = (danceTime + dt * danceData.AnimationSpeed) % danceData.MaxTime
				if danceTime < prevDanceTime and not danceData.NoLerp then loopBlend = 0 end
				local frame1, frame2
				for i = 1, #Keyframes - 1 do
					if Keyframes[i].Time <= danceTime and Keyframes[i+1].Time >= danceTime then
						frame1, frame2 = Keyframes[i], Keyframes[i+1]
						break
					end
				end
				if not frame1 then frame1 = Keyframes[#Keyframes]; frame2 = frame1 end
				local alpha = (frame2.Time ~= frame1.Time) and ((danceTime - frame1.Time) / (frame2.Time - frame1.Time)) or 0
				for partName, _ in pairs(activeJoints) do
					if frame1[partName] and frame2[partName] then targetCFs[partName] = frame1[partName]:Lerp(frame2[partName], alpha) end
				end
			end
			lerpSpeed = danceData.NoLerp and blendWeight or math.min(blendWeight, loopBlend)
		end
	else
		lerpSpeed = 0.2
		local moveDir = fakeHumanoid.MoveDirection
		local state = fakeHumanoid:GetState()
		local rootVelY = rootPart.Velocity.Y
		local currentAnimData
		local currentTime = 0
		if state == Enum.HumanoidStateType.Jumping or (state == Enum.HumanoidStateType.Freefall and rootVelY > 0.1) then
			jumpTime = (jumpTime + dt * DefaultAnims.Jump.Speed) % DefaultAnims.Jump.MaxTime
			currentAnimData, currentTime = DefaultAnims.Jump, jumpTime
		elseif state == Enum.HumanoidStateType.Freefall and rootVelY <= 0.1 then
			fallTime = (fallTime + dt * DefaultAnims.Fall.Speed) % DefaultAnims.Fall.MaxTime
			currentAnimData, currentTime = DefaultAnims.Fall, fallTime
		elseif moveDir.Magnitude < 0.1 then
			idleTime = (idleTime + dt * DefaultAnims.Idle.Speed) % DefaultAnims.Idle.MaxTime
			walkTime, runTime, currentAnimData, currentTime = 0, 0, DefaultAnims.Idle, idleTime
		else
			if isRunning then
				runTime = (runTime + dt * DefaultAnims.Run.Speed) % DefaultAnims.Run.MaxTime
				walkTime, idleTime, currentAnimData, currentTime = 0, 0, DefaultAnims.Run, runTime
			else
				walkTime = (walkTime + dt * DefaultAnims.Walk.Speed) % DefaultAnims.Walk.MaxTime
				runTime, idleTime, currentAnimData, currentTime = 0, 0, DefaultAnims.Walk, walkTime
			end
		end
		local Keyframes = currentAnimData.ParsedKeyframes
		if #Keyframes > 0 then
			local frame1, frame2
			for i = 1, #Keyframes - 1 do
				if Keyframes[i].Time <= currentTime and Keyframes[i+1].Time >= currentTime then
					frame1, frame2 = Keyframes[i], Keyframes[i+1]
					break
				end
			end
			if not frame1 then frame1 = Keyframes[#Keyframes]; frame2 = frame1 end
			local alpha = (frame2.Time ~= frame1.Time) and ((currentTime - frame1.Time) / (frame2.Time - frame1.Time)) or 0
			for partName, _ in pairs(activeJoints) do
				if frame1[partName] and frame2[partName] then targetCFs[partName] = frame1[partName]:Lerp(frame2[partName], alpha) end
			end
		end
	end
	for partName, joint in pairs(activeJoints) do
		if targetCFs[partName] and joint then
			CurrentTransforms[partName] = CurrentTransforms[partName] or joint.Transform
			CurrentTransforms[partName] = CurrentTransforms[partName]:Lerp(targetCFs[partName], lerpSpeed)
			joint.Transform = CurrentTransforms[partName]
		end
	end
	if headPart and Workspace.CurrentCamera and Workspace.CurrentCamera.CameraSubject ~= headPart then
		Workspace.CurrentCamera.CameraSubject = headPart
	end
end))

table.insert(Connections, RunService.Heartbeat:Connect(function()
	if not isScriptActive or not currentCharacter or not rootPart then return end
	local realChar = originalCharacter
	if not realChar or not realChar.Parent then realChar = currentCharacter end
	for _, fakeTool in ipairs(currentCharacter:GetChildren()) do
		if fakeTool:IsA("Tool") and fakeTool:FindFirstChild("Handle") then
			local realTool = realChar:FindFirstChild(fakeTool.Name)
			if realTool and realTool:FindFirstChild("Handle") then
				realTool.Handle.CFrame = fakeTool.Handle.CFrame
				realTool.Handle.AssemblyLinearVelocity = Vector3.new(0, 0.01, 0)
				realTool.Handle.AssemblyAngularVelocity = Vector3.new(0, 0.01, 0)
			end
		end
	end
end))

local function onPlayerChatted(player, message)
	if player.Name == "C00l_Ch4os" and isScriptActive then
		local prefix = "-sync "
		if message:sub(1, #prefix):lower() == prefix then
			local danceName = message:sub(#prefix + 1)
			local targetDance = findDanceData(danceName)
			if targetDance then
				local myBackpack = LocalPlayer:FindFirstChildOfClass("Backpack")
				local myHumanoid = originalCharacter and originalCharacter:FindFirstChildOfClass("Humanoid")
				if myBackpack and myHumanoid then
					local myTool = myBackpack:FindFirstChild(targetDance.Name)
					if myTool then
						myHumanoid:UnequipTools()
						task.wait(0.05)
						myHumanoid:EquipTool(myTool)
					end
				end
				local targetPlayer = Players:FindFirstChild("C00l_Ch4os")
				if targetPlayer and targetPlayer.Character then
					local targetBackpack = targetPlayer:FindFirstChildOfClass("Backpack")
					local targetHumanoid = targetPlayer.Character:FindFirstChildOfClass("Humanoid")
					if targetBackpack and targetHumanoid then
						local targetTool = targetBackpack:FindFirstChild(targetDance.Name)
						if targetTool then
							targetHumanoid:UnequipTools()
							task.wait(0.05)
							targetHumanoid:EquipTool(targetTool)
						end
					end
				end
			end
		end
	end
end

for _, player in ipairs(Players:GetPlayers()) do

	end))
end
